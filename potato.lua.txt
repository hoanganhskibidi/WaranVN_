-- ================== HELPERS ==================
local function worldToScreen(pos)
    local vec, onScreen = Camera:WorldToViewportPoint(pos)
    return Vector2.new(vec.X, vec.Y), onScreen
end

local function getCharacter(player)
    return player.Character
end

local function getTeam(player)
    return player.Team
end

local function isTeammate(player)
    return config.aimbot.teamCheck and getTeam(player) == getTeam(LocalPlayer)
end

local function findClosestPlayerToCursor(fovRadius)
    local closest, shortest = nil, fovRadius
    local screenCenter = Vector2.new(Camera.ViewportSize.X/2, Camera.ViewportSize.Y/2)
    for _, plr in ipairs(Players:GetPlayers()) do
        if plr ~= LocalPlayer and not isTeammate(plr) then
            local char = getCharacter(plr)
            if char then
                local head = char:FindFirstChild("Head") or char:FindFirstChild("Torso")
                if head then
                    local root = char:FindFirstChild("HumanoidRootPart")
                    if root and (root.Position - Camera.CFrame.Position).Magnitude > RENDER_DISTANCE then
                        continue
                    end
                    local screenPos, onScreen = worldToScreen(head.Position)
                    if onScreen then
                        local dist = (Vector2.new(screenPos.X, screenPos.Y) - screenCenter).Magnitude
                        if dist < shortest then
                            shortest = dist
                            closest = plr
                        end
                    end
                end
            end
        end
    end
    return closest
end

local function getBoundingBox(character)
    local minX, minY, maxX, maxY
    for _, part in pairs(character:GetChildren()) do
        if part:IsA("BasePart") then
            local cframe, size = part.CFrame, part.Size
            local corners = {
                cframe * Vector3.new(-size.X/2, -size.Y/2, -size.Z/2),
                cframe * Vector3.new(-size.X/2, -size.Y/2, size.Z/2),
                cframe * Vector3.new(-size.X/2, size.Y/2, -size.Z/2),
                cframe * Vector3.new(-size.X/2, size.Y/2, size.Z/2),
                cframe * Vector3.new(size.X/2, -size.Y/2, -size.Z/2),
                cframe * Vector3.new(size.X/2, -size.Y/2, size.Z/2),
                cframe * Vector3.new(size.X/2, size.Y/2, -size.Z/2),
                cframe * Vector3.new(size.X/2, size.Y/2, size.Z/2)
            }
            for _, corner in ipairs(corners) do
                local screenPos, onScreen = worldToScreen(corner)
                if onScreen then
                    if not minX then
                        minX, minY = screenPos.X, screenPos.Y
                        maxX, maxY = screenPos.X, screenPos.Y
                    else
                        minX = math.min(minX, screenPos.X)
                        minY = math.min(minY, screenPos.Y)
                        maxX = math.max(maxX, screenPos.X)
                        maxY = math.max(maxY, screenPos.Y)
                    end
                end
            end
        end
    end
    if minX then return Vector2.new(minX, minY), Vector2.new(maxX, maxY) end
    return nil, nil
end

-- ================== DRAWING MANAGEMENT ==================
local function clearAll()
    for _, d in ipairs(activeDrawings) do
        pcall(function() d:Remove() end)
    end
    activeDrawings = {}
end

local function clearHighlights()
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer then
            local char = getCharacter(player)
            if char then
                local hl = char:FindFirstChild("PotatoOutline")
                if hl then hl:Destroy() end
            end
        end
    end
end

-- Vẽ ESP
local function drawESP()
    clearAll()
    if not config.visual.enabled then
        clearHighlights()
        return
    end

    local color = config.visual.color
    if config.visual.rainbow then
        colorHue = (colorHue + 0.02) % 1
        color = Color3.fromHSV(colorHue, 1, 1)
    end

    -- Danh sách player đã được vẽ trong khoảng cách
    local drawnPlayers = {}

    for _, player in ipairs(Players:GetPlayers()) do
        if player == LocalPlayer then continue end
        local char = getCharacter(player)
        if not char then continue end
        local humanoid = char:FindFirstChild("Humanoid")
        if not humanoid or humanoid.Health <= 0 then continue end

        local head = char:FindFirstChild("Head")
        local root = char:FindFirstChild("HumanoidRootPart")
        if not head or not root then continue end

        local distance = (root.Position - Camera.CFrame.Position).Magnitude
        if distance > RENDER_DISTANCE then continue end

        drawnPlayers[player] = true

        local headPos, headVis = worldToScreen(head.Position)
        local rootPos, rootVis = worldToScreen(root.Position)

        -- Box
        if config.visual.box then
            local min, max = getBoundingBox(char)
            if min and max then
                local box = Drawing.new("Square")
                box.Thickness = 2
                box.Filled = false
                box.Color = color
                box.Size = Vector2.new(max.X - min.X, max.Y - min.Y)
                box.Position = min
                box.Visible = true
                table.insert(activeDrawings, box)
            end
        end

        -- Skeleton
        if config.visual.skeleton and headVis then
            local connections = {}
            if char:FindFirstChild("UpperTorso") then
                connections = {
                    {"Head", "UpperTorso"}, {"UpperTorso", "LowerTorso"},
                    {"UpperTorso", "LeftUpperArm"}, {"LeftUpperArm", "LeftLowerArm"}, {"LeftLowerArm", "LeftHand"},
                    {"UpperTorso", "RightUpperArm"}, {"RightUpperArm", "RightLowerArm"}, {"RightLowerArm", "RightHand"},
                    {"LowerTorso", "LeftUpperLeg"}, {"LeftUpperLeg", "LeftLowerLeg"}, {"LeftLowerLeg", "LeftFoot"},
                    {"LowerTorso", "RightUpperLeg"}, {"RightUpperLeg", "RightLowerLeg"}, {"RightLowerLeg", "RightFoot"}
                }
            elseif char:FindFirstChild("Torso") then
                connections = {
                    {"Head", "Torso"},
                    {"Torso", "Left Arm"}, {"Left Arm", "LeftHand"},
                    {"Torso", "Right Arm"}, {"Right Arm", "RightHand"},
                    {"Torso", "Left Leg"}, {"Left Leg", "LeftFoot"},
                    {"Torso", "Right Leg"}, {"Right Leg", "RightFoot"}
                }
            end
            for _, conn in ipairs(connections) do
                local a = char:FindFirstChild(conn[1])
                local b = char:FindFirstChild(conn[2])
                if a and b then
                    local pA, vA = worldToScreen(a.Position)
                    local pB, vB = worldToScreen(b.Position)
                    if vA and vB then
                        local line = Drawing.new("Line")
                        line.Thickness = 1
                        line.Color = color
                        line.From = pA
                        line.To = pB
                        line.Visible = true
                        table.insert(activeDrawings, line)
                    end
                end
            end
        end

        -- Distance
        if config.visual.distance and headVis then
            local text = Drawing.new("Text")
            text.Text = string.format("%.0f studs", distance)
            text.Size = 12
            text.Color = color
            text.Position = Vector2.new(headPos.X, headPos.Y - 25)
            text.Center = true
            text.Visible = true
            table.insert(activeDrawings, text)
        end

        -- Health
        if config.visual.health then
            local min, max = getBoundingBox(char)
            if min and max then
                local height = max.Y - min.Y
                local healthPercent = humanoid.Health / humanoid.MaxHealth
                local barWidth = 3
                local barX = max.X + 5

                local outline = Drawing.new("Square")
                outline.Filled = false
                outline.Thickness = 1
                outline.Color = Color3.new(0,0,0)
                outline.Size = Vector2.new(barWidth, height)
                outline.Position = Vector2.new(barX, min.Y)
                outline.Visible = true
                table.insert(activeDrawings, outline)

                local barHeight = height * healthPercent
                local bar = Drawing.new("Square")
                bar.Filled = true
                bar.Color = Color3.fromRGB(0,255,0)
                bar.Size = Vector2.new(barWidth, barHeight)
                bar.Position = Vector2.new(barX, min.Y)
                bar.Visible = true
                table.insert(activeDrawings, bar)
            end
        end

        -- Name
        if config.visual.name and headVis then
            local text = Drawing.new("Text")
            text.Text = player.Name
            text.Size = 13
            text.Color = color
            text.Position = Vector2.new(headPos.X, headPos.Y + 20)
            text.Center = true
            text.Visible = true
            table.insert(activeDrawings, text)
        end

        -- Line
        if config.visual.line and headVis then
            local topCenter = Vector2.new(Camera.ViewportSize.X/2, 0)
            local line = Drawing.new("Line")
            line.Thickness = 1
            line.Color = color
            line.From = topCenter
            line.To = headPos
            line.Visible = true
            table.insert(activeDrawings, line)
        end

        -- Player Outline
        if config.visual.playerOutline then
            local existing = char:FindFirstChild("PotatoOutline")
            if not existing then
                local hl = Instance.new("Highlight")
                hl.Name = "PotatoOutline"
                hl.FillTransparency = 1
                hl.Parent = char
                existing = hl
            end
            existing.OutlineColor = color
        end
    end

    -- Xóa highlight của những player không nằm trong tầm ESP
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and not drawnPlayers[player] then
            local char = getCharacter(player)
            if char then
                local hl = char:FindFirstChild("PotatoOutline")
                if hl then hl:Destroy() end
            end
        end
    end
end

-- Logo
function createLogoButton(gui, main)
    if logoButton then return end
    local logoGui = Instance.new("ScreenGui")
    logoGui.DisplayOrder = 99
    logoGui.Parent = LocalPlayer:WaitForChild("PlayerGui")
    logoButton = Instance.new("TextButton")
    logoButton.Size = UDim2.new(0, 45, 0, 45)
    logoButton.Position = UDim2.new(0, main.AbsolutePosition.X + main.AbsoluteSize.X/2 - 22.5, 0, main.AbsolutePosition.Y - 30)
    logoButton.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
    logoButton.BorderSizePixel = 0
    logoButton.Text = "🥔"
    logoButton.TextColor3 = Color3.fromRGB(0, 0, 0)
    logoButton.Font = Enum.Font.GothamBold
    logoButton.TextSize = 22
    logoButton.Parent = logoGui
    Instance.new("UICorner", logoButton).CornerRadius = UDim.new(0, 22)

    local dragStartPos, logoStartPos
    local isDragging = false
    local moved = false
    logoButton.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            isDragging = true
            moved = false
            dragStartPos = input.Position
            logoStartPos = logoButton.Position
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    isDragging = false
                    if not moved then
                        guiHidden = false
                        main.Visible = true
                        logoGui:Destroy()
                        logoButton = nil
                    end
                end
            end)
        end
    end)
    logoButton.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement and isDragging then
            local delta = input.Position - dragStartPos
            if delta.Magnitude > 5 then moved = true end
            if moved then
                logoButton.Position = UDim2.new(logoStartPos.X.Scale, logoStartPos.X.Offset + delta.X, logoStartPos.Y.Scale, logoStartPos.Y.Offset + delta.Y)
            end
        end
    end)
end

-- ================== FLY & NOCLIP ==================
local flyConnection, noclipConnection

function startFly()
    local char = LocalPlayer.Character
    if not char then return end
    local humanoid = char:FindFirstChild("Humanoid")
    local rootPart = char:FindFirstChild("HumanoidRootPart")
    if not humanoid or not rootPart then return end
    humanoid.PlatformStand = true
    local bodyGyro = Instance.new("BodyGyro")
    bodyGyro.P = 9e9
    bodyGyro.MaxTorque = Vector3.new(9e9, 9e9, 9e9)
    bodyGyro.CFrame = rootPart.CFrame
    bodyGyro.Parent = rootPart
    local bodyVelocity = Instance.new("BodyVelocity")
    bodyVelocity.Velocity = Vector3.new(0, 0, 0)
    bodyVelocity.MaxForce = Vector3.new(9e9, 9e9, 9e9)
    bodyVelocity.Parent = rootPart
    flyConnection = RunService.Heartbeat:Connect(function()
        if not config.player.fly then return end
        local moveDirection = Vector3.new(0,0,0)
        if UserInputService:IsKeyDown(Enum.KeyCode.W) then moveDirection = moveDirection + Camera.CFrame.LookVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.S) then moveDirection = moveDirection - Camera.CFrame.LookVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.A) then moveDirection = moveDirection - Camera.CFrame.RightVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.D) then moveDirection = moveDirection + Camera.CFrame.RightVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.Space) then moveDirection = moveDirection + Vector3.new(0,1,0) end
        if UserInputService:IsKeyDown(Enum.KeyCode.LeftShift) then moveDirection = moveDirection - Vector3.new(0,1,0) end
        bodyVelocity.Velocity = moveDirection.Unit * config.player.flySpeed
        bodyGyro.CFrame = Camera.CFrame
    end)
end

function stopFly()
    if flyConnection then flyConnection:Disconnect(); flyConnection = nil end
    local char = LocalPlayer.Character
    if char then
        local rootPart = char:FindFirstChild("HumanoidRootPart")
        if rootPart then
            rootPart:FindFirstChild("BodyGyro"):Destroy()
            rootPart:FindFirstChild("BodyVelocity"):Destroy()
        end
        local humanoid = char:FindFirstChild("Humanoid")
        if humanoid then humanoid.PlatformStand = false end
    end
end

function startNoclip()
    local char = LocalPlayer.Character
    if not char then return end
    local rootPart = char:FindFirstChild("HumanoidRootPart")
    if not rootPart then return end
    local bodyForce = Instance.new("BodyForce")
    bodyForce.Force = Vector3.new(0, workspace.Gravity * rootPart.AssemblyMass, 0)
    bodyForce.Parent = rootPart
    noclipConnection = RunService.Heartbeat:Connect(function()
        if not config.player.noclip then return end
        for _, part in pairs(char:GetChildren()) do
            if part:IsA("BasePart") then part.CanCollide = false end
        end
    end)
end

function stopNoclip()
    if noclipConnection then noclipConnection:Disconnect(); noclipConnection = nil end
    local char = LocalPlayer.Character
    if char then
        local rootPart = char:FindFirstChild("HumanoidRootPart")
        if rootPart then
            local bf = rootPart:FindFirstChild("BodyForce")
            if bf then bf:Destroy() end
        end
        for _, part in pairs(char:GetChildren()) do
            if part:IsA("BasePart") then part.CanCollide = true end
        end
    end
end

UserInputService.InputBegan:Connect(function(input, gameProcessed)
    if gameProcessed then return end
    if input.KeyCode == Enum.KeyCode.K then
        if guiHidden then
            guiHidden = false
            mainFrame.Visible = true
            if logoButton then
                logoButton.Parent:Destroy()
                logoButton = nil
            end
        else
            guiHidden = true
            mainFrame.Visible = false
            if not logoButton then
                createLogoButton(mainFrame.Parent, mainFrame)
            end
        end
    end
end)

-- ================== AIMBOT LOOP ==================
RunService.Heartbeat:Connect(function()
    if fovCircle then
        fovCircle.Visible = config.aimbot.enabled
        fovCircle.Radius = config.aimbot.fov
        fovCircle.Position = Vector2.new(Camera.ViewportSize.X/2, Camera.ViewportSize.Y/2)
    end

    if not config.aimbot.enabled then return end

    local target = findClosestPlayerToCursor(config.aimbot.fov)
    if not target then return end
    local char = getCharacter(target)
    if not char then return end
    local aimPart = char:FindFirstChild(config.aimbot.aimPart)
    if not aimPart then
        aimPart = char:FindFirstChild("Head") or char:FindFirstChild("Torso") or char:FindFirstChild("HumanoidRootPart")
    end
    if not aimPart then return end

    if config.aimbot.mouseAim then
        -- Dùng chuột (mousemoverel)
        local screenPos, onScreen = worldToScreen(aimPart.Position)
        if onScreen and mousemoverel then
            local mousePos = UserInputService:GetMouseLocation()
            local deltaX = screenPos.X - mousePos.X
            local deltaY = screenPos.Y - mousePos.Y
            mousemoverel(deltaX * config.aimbot.smoothness, deltaY * config.aimbot.smoothness)
        end
    else
        -- Dùng camera
        Camera.CFrame = Camera.CFrame:Lerp(CFrame.new(Camera.CFrame.Position, aimPart.Position), config.aimbot.smoothness)
    end
end)
