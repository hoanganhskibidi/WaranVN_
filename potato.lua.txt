-- ================== HELPER FUNCTIONS ==================
local function worldToScreen(position)
    local vector, onScreen = Camera:WorldToViewportPoint(position)
    return Vector2.new(vector.X, vector.Y), onScreen
end

local function getTeam(player) return player.Team end
local function isTeammate(player)
    return (config.aimbot.teamCheck or config.visual.teamCheck) and getTeam(player) == getTeam(LocalPlayer)
end

local function getCharacter(player)
    if player.Character and player.Character.Parent then return player.Character end
    local char = workspace:FindFirstChild(player.Name)
    if char and (char:FindFirstChild("Humanoid") or char:FindFirstChild("HumanoidRootPart")) then return char end
    return nil
end

local function getCorePart(character)
    return character:FindFirstChild("HumanoidRootPart") or character:FindFirstChild("Torso") or character:FindFirstChild("UpperTorso") or character:FindFirstChildWhichIsA("BasePart")
end

local function getTopPart(character)
    return character:FindFirstChild("Head") or character:FindFirstChild("UpperTorso") or character:FindFirstChild("Torso") or character:FindFirstChildWhichIsA("BasePart")
end

local function findClosestPlayerToCursor(fovRadius)
    local closestPlayer = nil
    local shortestDistance = fovRadius
    local screenCenter = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y / 2)

    for _, otherPlayer in ipairs(Players:GetPlayers()) do
        if otherPlayer ~= LocalPlayer and not isTeammate(otherPlayer) then
            local character = getCharacter(otherPlayer)
            if character then
                local corePart = getCorePart(character)
                local aimPart = character:FindFirstChild(config.aimbot.aimPart) or getTopPart(character) or corePart
                if aimPart and corePart and (corePart.Position - Camera.CFrame.Position).Magnitude <= RENDER_DISTANCE then
                    local screenPosition, onScreen = worldToScreen(aimPart.Position)
                    if onScreen then
                        local distance = (Vector2.new(screenPosition.X, screenPosition.Y) - screenCenter).Magnitude
                        if distance < shortestDistance then
                            shortestDistance = distance;
                            closestPlayer = otherPlayer
                        end
                    end
                end
            end
        end
    end
    return closestPlayer
end

-- ================== DRAWING CACHE & CLEANUP ==================
local espCache = {}
local function hideDrawings(cache)
    if not cache then return end
    if cache.box then cache.box.Visible = false end
    if cache.healthOutline then cache.healthOutline.Visible = false end
    if cache.healthBar then cache.healthBar.Visible = false end
    if cache.tracerLine then cache.tracerLine.Visible = false end
    if cache.skeletonLines then
        for _, l in ipairs(cache.skeletonLines) do l.Visible = false end
    end
end

local function removePlayerCache(plr)
    if activeTags[plr] then pcall(function() activeTags[plr]:Destroy() end); activeTags[plr] = nil end
    if espCache[plr] then
        pcall(function()
            espCache[plr].box:Remove()
            espCache[plr].healthOutline:Remove()
            espCache[plr].healthBar:Remove()
            espCache[plr].tracerLine:Remove()
            for _, l in ipairs(espCache[plr].skeletonLines) do l:Remove() end
        end)
        espCache[plr] = nil
    end
end

Players.PlayerRemoving:Connect(removePlayerCache)

local function cleanStaleData()
    for plr, _ in pairs(espCache) do if not plr or not plr.Parent or not Players:FindFirstChild(plr.Name) then removePlayerCache(plr) end end
    for plr, _ in pairs(activeTags) do if not plr or not plr.Parent or not Players:FindFirstChild(plr.Name) then removePlayerCache(plr) end end
end

local function getBoundingBox(character)
    local hrp = character:FindFirstChild("HumanoidRootPart")
    if not hrp then return nil, nil end
    local hrpPos, onScreen = Camera:WorldToViewportPoint(hrp.Position)
    if not onScreen then return nil, nil end
    
    local head = character:FindFirstChild("Head")
    local topWorld = head and (head.Position + Vector3.new(0, 1.5, 0)) or (hrp.Position + Vector3.new(0, 2.5, 0))
    local bottomWorld = hrp.Position - Vector3.new(0, 3, 0)
    
    local topScreen, topOn = Camera:WorldToViewportPoint(topWorld)
    local bottomScreen, bottomOn = Camera:WorldToViewportPoint(bottomWorld)
    if not topOn and not bottomOn then return nil, nil end
    
    local height = math.abs(topScreen.Y - bottomScreen.Y)
    local width = height / 1.55 
    return Vector2.new(hrpPos.X - width / 2, topScreen.Y), Vector2.new(width, height)
end

-- ================== SILENT AIM SYSTEM ==================
local silentAimConnection = nil
local function findClosestEnemyInFOV(fovRadius)
    local closestEnemy = nil
    local shortestDistance = fovRadius
    local screenCenter = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y / 2)

    for _, otherPlayer in ipairs(Players:GetPlayers()) do
        if otherPlayer ~= LocalPlayer and not isTeammate(otherPlayer) then
            local character = getCharacter(otherPlayer)
            if character then
                local topPart = getTopPart(character)
                local corePart = getCorePart(character)
                if topPart and corePart and (corePart.Position - Camera.CFrame.Position).Magnitude <= RENDER_DISTANCE then
                    local screenPosition, onScreen = Camera:WorldToViewportPoint(topPart.Position)
                    if onScreen then
                        local distance = (Vector2.new(screenPosition.X, screenPosition.Y) - screenCenter).Magnitude
                        if distance < shortestDistance then shortestDistance = distance; closestEnemy = otherPlayer end
                    end
                end
            end
        end
    end
    return closestEnemy
end

local function getGunTool()
    local character = LocalPlayer.Character
    if not character then return nil end
    for _, tool in ipairs(character:GetChildren()) do
        if tool:IsA("Tool") and (tool:FindFirstChild("Handle") or tool:GetAttribute("IsGun")) then return tool end
    end
    return nil
end

function setupSilentAim()
    if silentAimConnection then silentAimConnection:Disconnect(); silentAimConnection = nil end
    if not config.silentAim.enabled then return end
    local tool = getGunTool()
    if not tool then return end

    local remote = tool:FindFirstChild("RemoteEvent") or tool:FindFirstChild("ShootRemote") or tool:FindFirstChild("Fire")
    if not remote then return end

    silentAimConnection = remote.OnServerEvent:Connect(function(player, ...)
        if player ~= LocalPlayer then return end
        if config.silentAim.autoDetectGun and not getGunTool() then return end
        if config.silentAim.randomDelay then task.wait(math.random(50, 150) / 1000) end
        if math.random(1, 100) <= config.silentAim.missChance then return end
        
        local target = findClosestEnemyInFOV(config.silentAim.fov)
        if not target then return end
        local character = getCharacter(target)
        if not character then return end
        local targetPart = getTopPart(character)
        if not targetPart then return end

        local args = {...}
        if #args >= 1 and typeof(args[1]) == "Vector3" then
            args[1] = targetPart.Position
            pcall(function() remote:FireServer(unpack(args)) end)
            return nil
        end
    end)
end

local function drawESP()
    cleanStaleData()
    local renderedTags = {}

    if not config.visual.enabled then
        clearHighlights()
        for _, tag in pairs(activeTags) do if tag then tag.Enabled = false end end
        for _, cache in pairs(espCache) do hideDrawings(cache) end
        return
    end

    local baseColor = config.visual.color
    if config.visual.rainbow then
        colorHue = (colorHue + 0.005 * config.visual.rainbowSpeed) % 1
        baseColor = Color3.fromHSV(colorHue, 0.7, 1)
    end

    local drawnPlayers = {}

    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer then
            pcall(function()
                if not espCache[player] then
                    espCache[player] = {
                        box = Drawing.new("Square"),
                        healthOutline = Drawing.new("Square"),
                        healthBar = Drawing.new("Square"),
                        tracerLine = Drawing.new("Line"),
                        skeletonLines = {}
                    }
                    local c = espCache[player]
                    c.box.Thickness = 1; c.box.Filled = false
                    c.healthOutline.Thickness = 1; c.healthOutline.Filled = false; c.healthOutline.Color = Color3.new(0,0,0)
                    c.healthBar.Filled = true
                    c.tracerLine.Thickness = 1
                    for i = 1, 15 do c.skeletonLines[i] = Drawing.new("Line"); c.skeletonLines[i].Thickness = 1 end
                end

                local cache = espCache[player]
                if isTeammate(player) then hideDrawings(cache); return end
                local character = getCharacter(player)
                if not character or not character:IsDescendantOf(workspace) then hideDrawings(cache); return end
                local corePart = getCorePart(character)
                local topPart = getTopPart(character)
                if not corePart or not topPart then hideDrawings(cache); return end

                local distanceFromCamera = (corePart.Position - Camera.CFrame.Position).Magnitude
                if distanceFromCamera > RENDER_DISTANCE then hideDrawings(cache); return end

                local humanoid = character:FindFirstChildOfClass("Humanoid")
                if not humanoid or humanoid.Health <= 0 then
                    hideDrawings(cache)
                    if activeTags[player] then activeTags[player].Enabled = false end
                    return
                end

                drawnPlayers[player] = true
                local topScreenPos, topVisible = worldToScreen(topPart.Position)
                local minVector, sizeVector = getBoundingBox(character)

                ---- BOX ----
                if config.visual.box and minVector and sizeVector then
                    cache.box.Color = baseColor; cache.box.Size = sizeVector; cache.box.Position = minVector; cache.box.Visible = true
                else cache.box.Visible = false end

                ---- SIDE HEALTH BAR ----
                if config.visual.healthBar and minVector and sizeVector and humanoid then
                    local fullHeight = sizeVector.Y
                    local maxHealth = humanoid.MaxHealth > 0 and humanoid.MaxHealth or 100
                    local healthPercent = math.clamp(humanoid.Health / maxHealth, 0, 1)
                    local barWidth = 2; local barX = minVector.X - 5 

                    cache.healthOutline.Size = Vector2.new(barWidth, fullHeight)
                    cache.healthOutline.Position = Vector2.new(barX, minVector.Y); cache.healthOutline.Visible = true

                    local barHeight = fullHeight * healthPercent
                    cache.healthBar.Color = Color3.fromRGB(255 - (255 * healthPercent), 255 * healthPercent, 0) 
                    cache.healthBar.Size = Vector2.new(barWidth, barHeight)
                    cache.healthBar.Position = Vector2.new(barX, minVector.Y + (fullHeight - barHeight)); cache.healthBar.Visible = true
                else cache.healthOutline.Visible = false; cache.healthBar.Visible = false end

                ---- SKELETON ----
                local skelIdx = 1
                if config.visual.skeleton and topVisible then
                    local connections = {}
                    if character:FindFirstChild("UpperTorso") then
                        connections = {
                            {"Head", "UpperTorso"}, {"UpperTorso", "LowerTorso"},
                            {"UpperTorso", "LeftUpperArm"}, {"LeftUpperArm", "LeftLowerArm"}, {"LeftLowerArm", "LeftHand"},
                            {"UpperTorso", "RightUpperArm"}, {"RightUpperArm", "RightLowerArm"}, {"RightLowerArm", "RightHand"},
                            {"LowerTorso", "LeftUpperLeg"}, {"LeftUpperLeg", "LeftLowerLeg"}, {"LeftLowerLeg", "LeftFoot"},
                            {"LowerTorso", "RightUpperLeg"}, {"RightUpperLeg", "RightLowerLeg"}, {"RightLowerLeg", "RightFoot"}
                        }
                    elseif character:FindFirstChild("Torso") then
                        connections = {
                            {"Head", "Torso"}, {"Torso", "Left Arm"}, {"Left Arm", "LeftHand"},
                            {"Torso", "Right Arm"}, {"Right Arm", "RightHand"},
                            {"Torso", "Left Leg"}, {"Left Leg", "LeftFoot"},
                            {"Torso", "Right Leg"}, {"Right Leg", "RightFoot"}
                        }
                    end
                    for _, connection in ipairs(connections) do
                        local partA = character:FindFirstChild(connection[1]); local partB = character:FindFirstChild(connection[2])
                        if partA and partB then
                            local screenPosA, visibleA = worldToScreen(partA.Position); local screenPosB, visibleB = worldToScreen(partB.Position)
                            if visibleA and visibleB then
                                local line = cache.skeletonLines[skelIdx]
                                line.Color = baseColor; line.From = screenPosA; line.To = screenPosB; line.Visible = true; skelIdx = skelIdx + 1
                            end
                        end
                    end
                end
                for i = skelIdx, 15 do cache.skeletonLines[i].Visible = false end

                ---- HYBRID INFO TAG ----
                if config.visual.infoTag and topVisible then
                    local tag = getOrCreateTag(player, character)
                    if tag and tag:FindFirstChild("MainFrame") then
                        local hpValue = math.floor(humanoid.Health)
                        tag.MainFrame.HPBox.HPText.Text = tostring(hpValue)
                        tag.MainFrame.NameText.Text = player.DisplayName or player.Name
                        tag.MainFrame.DistText.Text = string.format("%.0fm", distanceFromCamera)
                        
                        tag.Enabled = true
                        renderedTags[player] = true
                    end
                else
                    if activeTags[player] then activeTags[player].Enabled = false end
                end

                ---- TRACER LINE (TỪ TRÊN XUỐNG DƯỚI ĐẦU PLAYER) ----
                if config.visual.line and topVisible then
                    cache.tracerLine.Color = baseColor; 
                    -- Tọa độ Y = 0 để line bắt đầu từ Cạnh Trên Cùng giữa màn hình (Top Center)
                    cache.tracerLine.From = Vector2.new(Camera.ViewportSize.X / 2, 0); 
                    cache.tracerLine.To = topScreenPos; 
                    cache.tracerLine.Visible = true
                else cache.tracerLine.Visible = false end

                ---- OUTLINE ----
                if config.visual.playerOutline then
                    local hl = character:FindFirstChild("PotatoOutline")
                    if not hl then hl = Instance.new("Highlight"); hl.Name = "PotatoOutline"; hl.FillTransparency = 1; hl.Parent = character end
                    hl.OutlineColor = baseColor
                end
            end)
        end
    end

    for plr, tag in pairs(activeTags) do if not renderedTags[plr] and tag then tag.Enabled = false end end
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and not drawnPlayers[player] then
            local character = getCharacter(player)
            if character and character:FindFirstChild("PotatoOutline") then pcall(function() character.PotatoOutline:Destroy() end) end
            if espCache[player] then hideDrawings(espCache[player]) end
        end
    end
end

        local function updateValue(val)
            val = math.clamp(val, minVal, maxVal)
            val = math.floor(val * 100 + 0.5) / 100
            currentVal = val; valBox.Text = tostring(val)
            fill.Size = UDim2.new((val - minVal) / (maxVal - minVal), 0, 1, 0)
            callback(val)
        end

            local num = tonumber(valBox.Text)
            if num then updateValue(num) else valBox.Text = tostring(currentVal) end
        end)

        track.InputBegan:Connect(function(input)
            if input.UserInputType == Enum.UserInputType.MouseButton1 then
                draggingSlider = true
                local relX = math.clamp((input.Position.X - track.AbsolutePosition.X) / track.AbsoluteSize.X, 0, 1)

                updateValue(minVal + relX * (maxVal - minVal))
            end
        end)
    end

    Instance.new("UICorner", logoButton).CornerRadius = UDim.new(0, 8)
    Instance.new("UIStroke", logoButton).Color = Theme.Border

    local dragToggle = false
    local dragStart, startPos = nil, nil
    local dragMoved = false

    logoButton.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragToggle = true; dragMoved = false; dragStart = input.Position; startPos = logoButton.Position
        end
    end)
    logoButton.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement and dragToggle then
            local delta = input.Position - dragStart
            if delta.Magnitude > 3 then
                dragMoved = true
                logoButton.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
            end
        end
    end)
    logoButton.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 and dragToggle then
            dragToggle = false
            if not dragMoved then
                guiHidden = false; mainWindow.Visible = true
                if logoButton and logoButton.Parent then logoButton.Parent:Destroy() end
                logoButton = nil
            end
        end
    end)
end

-- ================== INTEGRATED INERTIAL FLY ENGINE (LERPING) ==================
local flyConnection = nil
function startFly()
    local char = LocalPlayer.Character
    if not char then return end
    local root = char:FindFirstChild("HumanoidRootPart")
    local hum = char:FindFirstChildOfClass("Humanoid")
    if not root or not hum then return end
    
    stopFly()
    
    hum.PlatformStand = true
    
    local bg = Instance.new("BodyGyro")
    bg.P = 9e4
    bg.MaxTorque = Vector3.new(9e9, 9e9, 9e9)
    bg.CFrame = root.CFrame
    bg.Parent = root
    
    local bv = Instance.new("BodyVelocity")
    bv.MaxForce = Vector3.new(9e9, 9e9, 9e9)
    bv.Velocity = Vector3.new(0, 0, 0)
    bv.Parent = root

    flyConnection = RunService.RenderStepped:Connect(function()
        if not config.player.fly or not char.Parent or not root or not hum then 
            stopFly()
            return 
        end
        
        local speed = config.player.flySpeed
        if hum.MoveDirection.Magnitude > 0 then
            local flyDir = (hum.MoveDirection + Vector3.new(0, Camera.CFrame.LookVector.Y * 1.5, 0)).Unit
            bv.Velocity = bv.Velocity:Lerp(flyDir * speed, 0.1)
            bg.CFrame = CFrame.new(root.Position, root.Position + bv.Velocity)
        else
            bv.Velocity = bv.Velocity:Lerp(Vector3.new(0, 0, 0), 0.1)
            bg.CFrame = bg.CFrame:Lerp(CFrame.new(root.Position, root.Position + Camera.CFrame.LookVector * Vector3.new(1, 0, 1)), 0.1)
        end
    end)
end

function stopFly()
    if flyConnection then flyConnection:Disconnect(); flyConnection = nil end
    local char = LocalPlayer.Character
    if char then
        local hum = char:FindFirstChildOfClass("Humanoid")
        if hum then hum.PlatformStand = false end
        local root = char:FindFirstChild("HumanoidRootPart")
        if root then
            if root:FindFirstChild("BodyGyro") then root.BodyGyro:Destroy() end
            if root:FindFirstChild("BodyVelocity") then root.BodyVelocity:Destroy() end
        end
    end
end

function startNoclip()
    local char = LocalPlayer.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return end
    local bf = Instance.new("BodyForce"); bf.Force = Vector3.new(0, workspace.Gravity * char.HumanoidRootPart.AssemblyMass, 0); bf.Parent = char.HumanoidRootPart
    noclipConnection = RunService.Heartbeat:Connect(function()
        if not config.player.noclip then return end
        for _, part in pairs(char:GetChildren()) do if part:IsA("BasePart") then part.CanCollide = false end end
    end)
end

function stopNoclip()
    if noclipConnection then noclipConnection:Disconnect(); noclipConnection = nil end
    local char = LocalPlayer.Character
    if char then
        if char:FindFirstChild("HumanoidRootPart") and char.HumanoidRootPart:FindFirstChild("BodyForce") then char.HumanoidRootPart.BodyForce:Destroy() end
        for _, part in pairs(char:GetChildren()) do if part:IsA("BasePart") then part.CanCollide = true end end
    end
end

        end
    else
        Camera.CFrame = Camera.CFrame:Lerp(CFrame.new(Camera.CFrame.Position, aimPart.Position), config.aimbot.smoothness)
    end
end)
