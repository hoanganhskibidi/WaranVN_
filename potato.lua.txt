-- ================== HELPER FUNCTIONS ==================
local function worldToScreen(position)
    local vector, onScreen = Camera:WorldToViewportPoint(position)
    return Vector2.new(vector.X, vector.Y), onScreen
end

local function getCharacter(player)
    return player.Character
end

local function getTeam(player)
    return player.Team
end

local function isTeammate(player)
    return (config.aimbot.teamCheck or config.visual.teamCheck) and getTeam(player) == getTeam(LocalPlayer)
end

local function findClosestPlayerToCursor(fovRadius)
    local closestPlayer = nil
    local shortestDistance = fovRadius
    local screenCenter = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y / 2)

    for _, otherPlayer in ipairs(Players:GetPlayers()) do
        if otherPlayer ~= LocalPlayer and not isTeammate(otherPlayer) then
            local character = getCharacter(otherPlayer)
            if character then
                local head = character:FindFirstChild("Head") or character:FindFirstChild("Torso")
                if head then
                    local rootPart = character:FindFirstChild("HumanoidRootPart")
                    if rootPart and (rootPart.Position - Camera.CFrame.Position).Magnitude > RENDER_DISTANCE then
                        continue
                    end
                    local screenPosition, onScreen = worldToScreen(head.Position)
                    if onScreen then
                        local distance = (Vector2.new(screenPosition.X, screenPosition.Y) - screenCenter).Magnitude
                        if distance < shortestDistance then
                            shortestDistance = distance
                            closestPlayer = otherPlayer
                        end
                    end
                end
            end
        end
    end

    return closestPlayer
end

-- Cache for getBoundingBox to avoid recalculating every frame for the same character
local boundingBoxCache = {}
local function getBoundingBox(character)
    -- Return cached result if available
    if boundingBoxCache[character] then
        return boundingBoxCache[character][1], boundingBoxCache[character][2]
    end

    local minX, minY, maxX, maxY
    for _, part in pairs(character:GetChildren()) do
        if part:IsA("BasePart") then
            local cframe = part.CFrame
            local size = part.Size
            -- 8 corners of the part
            local corners = {
                cframe * Vector3.new(-size.X/2, -size.Y/2, -size.Z/2),
                cframe * Vector3.new(-size.X/2, -size.Y/2, size.Z/2),
                cframe * Vector3.new(-size.X/2, size.Y/2, -size.Z/2),
                cframe * Vector3.new(-size.X/2, size.Y/2, size.Z/2),
                cframe * Vector3.new(size.X/2, -size.Y/2, -size.Z/2),
                cframe * Vector3.new(size.X/2, -size.Y/2, size.Z/2),
                cframe * Vector3.new(size.X/2, size.Y/2, -size.Z/2),
                cframe * Vector3.new(size.X/2, size.Y/2, size.Z/2)
            }
            for _, corner in ipairs(corners) do
                local screenPos, onScreen = worldToScreen(corner)
                if onScreen then
                    if not minX then
                        minX = screenPos.X
                        minY = screenPos.Y
                        maxX = screenPos.X
                        maxY = screenPos.Y
                    else
                        minX = math.min(minX, screenPos.X)
                        minY = math.min(minY, screenPos.Y)
                        maxX = math.max(maxX, screenPos.X)
                        maxY = math.max(maxY, screenPos.Y)
                    end
                end
            end
        end
    end

    if minX then
        local minVector = Vector2.new(minX, minY)
        local maxVector = Vector2.new(maxX, maxY)
        boundingBoxCache[character] = {minVector, maxVector}
        return minVector, maxVector
    end

    boundingBoxCache[character] = {nil, nil}
    return nil, nil
end

-- ================== SILENT AIM SYSTEM ==================
local silentAimConnection = nil

local function findClosestEnemyInFOV(fovRadius)
    local closestEnemy = nil
    local shortestDistance = fovRadius
    local screenCenter = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y / 2)

    for _, otherPlayer in ipairs(Players:GetPlayers()) do
        if otherPlayer ~= LocalPlayer and not (config.aimbot.teamCheck and otherPlayer.Team == LocalPlayer.Team) then
            local character = otherPlayer.Character
            if character and character:FindFirstChild("Humanoid") and character.Humanoid.Health > 0 then
                local head = character:FindFirstChild("Head")
                if head then
                    local rootPart = character:FindFirstChild("HumanoidRootPart")
                    if rootPart and (rootPart.Position - Camera.CFrame.Position).Magnitude <= RENDER_DISTANCE then
                        local screenPosition, onScreen = Camera:WorldToViewportPoint(head.Position)
                        if onScreen then
                            local distance = (Vector2.new(screenPosition.X, screenPosition.Y) - screenCenter).Magnitude
                            if distance < shortestDistance then
                                shortestDistance = distance
                                closestEnemy = otherPlayer
                            end
                        end
                    end
                end
            end
        end
    end

    return closestEnemy
end

local function getGunTool()
    local character = LocalPlayer.Character
    if not character then
        return nil
    end
    for _, tool in ipairs(character:GetChildren()) do
        if tool:IsA("Tool") and (tool:FindFirstChild("Handle") or tool:GetAttribute("IsGun")) then
            return tool
        end
    end
    return nil
end

function setupSilentAim()
    -- Disconnect previous connection
    if silentAimConnection then
        silentAimConnection:Disconnect()
        silentAimConnection = nil
    end

    if not config.silentAim.enabled then
        return
    end

    local tool = getGunTool()
    if not tool then
        return
    end

    -- Try to find the remote event used for shooting
    local remote = tool:FindFirstChild("RemoteEvent") or tool:FindFirstChild("ShootRemote") or tool:FindFirstChild("Fire")
    if not remote then
        return
    end

    -- Hook into the remote's OnServerEvent to intercept shooting
    silentAimConnection = remote.OnServerEvent:Connect(function(player, ...)
        if player ~= LocalPlayer then
            return
        end
        -- Check if still holding a gun (if autoDetectGun is enabled)
        if config.silentAim.autoDetectGun and not getGunTool() then
            return
        end
        -- Apply random delay
        if config.silentAim.randomDelay then
            wait(math.random(50, 150) / 1000)
        end
        -- Miss chance simulation
        if math.random(1, 100) <= config.silentAim.missChance then
            return
        end
        -- Find closest enemy in silent FOV
        local target = findClosestEnemyInFOV(config.silentAim.fov)
        if not target or not target.Character or not target.Character:FindFirstChild("Head") then
            return
        end
        local head = target.Character.Head

        -- Modify the arguments: replace the first Vector3 argument (assumed to be target position) with head position
        local args = {...}
        if #args >= 1 and typeof(args[1]) == "Vector3" then
            args[1] = head.Position
            -- Fire the remote again with modified arguments
            pcall(function()
                remote:FireServer(unpack(args))
            end)
            -- Prevent the original event from being processed further
            return nil
        end
    end)
end

-- Re‑setup silent aim when the local player's character changes or tools are added/removed
LocalPlayer.CharacterAdded:Connect(function(character)
    character.ChildAdded:Connect(function(child)
        if config.silentAim.enabled and child:IsA("Tool") then
            setupSilentAim()
        end
    end)
    character.ChildRemoved:Connect(function(child)
        if child:IsA("Tool") then
            setupSilentAim()
        end
    end)
end)

-- ================== VISUAL ESP ==================
local function clearAllDrawings()
    for _, drawing in ipairs(activeDrawings) do
        pcall(function() drawing:Remove() end)
    end
    activeDrawings = {}
end

local function clearHighlights()
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer then
            local character = getCharacter(player)
            if character then
                local highlight = character:FindFirstChild("PotatoOutline")
                if highlight then
                    highlight:Destroy()
                end
            end
        end
    end
end

local function drawESP()
    -- Clear cache every frame
    boundingBoxCache = {}
    clearAllDrawings()

    -- If ESP master switch is off, also clear all highlights and return
    if not config.visual.enabled then
        clearHighlights()
        return
    end

    -- Determine color
    local currentColor = config.visual.color
    if config.visual.rainbow then
        colorHue = (colorHue + 0.005 * config.visual.rainbowSpeed) % 1
        currentColor = Color3.fromHSV(colorHue, 1, 1)
    end

    local drawnPlayers = {}

    for _, player in ipairs(Players:GetPlayers()) do
        -- Skip local player and teammates if teamCheck is on
        if player == LocalPlayer or isTeammate(player) then
            continue
        end

        local character = getCharacter(player)
        if not character then
            continue
        end

        local humanoid = character:FindFirstChild("Humanoid")
        if not humanoid or humanoid.Health <= 0 then
            continue
        end

        local head = character:FindFirstChild("Head")
        local rootPart = character:FindFirstChild("HumanoidRootPart")
        if not head or not rootPart then
            continue
        end

        -- Distance check (render distance)
        local distanceFromCamera = (rootPart.Position - Camera.CFrame.Position).Magnitude
        if distanceFromCamera > RENDER_DISTANCE then
            continue
        end

        drawnPlayers[player] = true

        local headScreenPos, headVisible = worldToScreen(head.Position)
        local rootScreenPos, rootVisible = worldToScreen(rootPart.Position)

        -- Stop drawing if neither head nor root is on screen (performance)
        if not headVisible and not rootVisible then
            continue
        end

        ---- BOX ----
        if config.visual.box then
            local minVector, maxVector = getBoundingBox(character)
            if minVector and maxVector then
                local box = Drawing.new("Square")
                box.Thickness = 2
                box.Filled = false
                box.Color = currentColor
                box.Size = Vector2.new(maxVector.X - minVector.X, maxVector.Y - minVector.Y)
                box.Position = minVector
                box.Visible = true
                table.insert(activeDrawings, box)
            end
        end

        ---- SKELETON (R15 and R6) ----
        if config.visual.skeleton and headVisible then
            local connections = {}
            if character:FindFirstChild("UpperTorso") then
                -- R15 rig
                connections = {
                    {"Head", "UpperTorso"}, {"UpperTorso", "LowerTorso"},
                    {"UpperTorso", "LeftUpperArm"}, {"LeftUpperArm", "LeftLowerArm"}, {"LeftLowerArm", "LeftHand"},
                    {"UpperTorso", "RightUpperArm"}, {"RightUpperArm", "RightLowerArm"}, {"RightLowerArm", "RightHand"},
                    {"LowerTorso", "LeftUpperLeg"}, {"LeftUpperLeg", "LeftLowerLeg"}, {"LeftLowerLeg", "LeftFoot"},
                    {"LowerTorso", "RightUpperLeg"}, {"RightUpperLeg", "RightLowerLeg"}, {"RightLowerLeg", "RightFoot"}
                }
            elseif character:FindFirstChild("Torso") then
                -- R6 rig
                connections = {
                    {"Head", "Torso"},
                    {"Torso", "Left Arm"}, {"Left Arm", "LeftHand"},
                    {"Torso", "Right Arm"}, {"Right Arm", "RightHand"},
                    {"Torso", "Left Leg"}, {"Left Leg", "LeftFoot"},
                    {"Torso", "Right Leg"}, {"Right Leg", "RightFoot"}
                }
            end
            for _, connection in ipairs(connections) do
                local partA = character:FindFirstChild(connection[1])
                local partB = character:FindFirstChild(connection[2])
                if partA and partB then
                    local screenPosA, visibleA = worldToScreen(partA.Position)
                    local screenPosB, visibleB = worldToScreen(partB.Position)
                    if visibleA and visibleB then
                        local line = Drawing.new("Line")
                        line.Thickness = 1
                        line.Color = currentColor
                        line.From = screenPosA
                        line.To = screenPosB
                        line.Visible = true
                        table.insert(activeDrawings, line)
                    end
                end
            end
        end

        ---- DISTANCE ----
        if config.visual.distance and headVisible then
            local distanceText = Drawing.new("Text")
            distanceText.Text = string.format("%.0f studs", distanceFromCamera)
            distanceText.Size = 12
            distanceText.Color = currentColor
            distanceText.Position = Vector2.new(headScreenPos.X, headScreenPos.Y - 25)
            distanceText.Center = true
            distanceText.Visible = true
            table.insert(activeDrawings, distanceText)
        end

        ---- HEALTH BAR (proportional to character height) ----
        if config.visual.health and headVisible then
            local minVector, maxVector = getBoundingBox(character)
            if minVector and maxVector then
                local fullHeight = maxVector.Y - minVector.Y
                local healthPercent = humanoid.Health / humanoid.MaxHealth
                local barWidth = 3
                local barX = maxVector.X + 5

                -- Outline (black border)
                local healthOutline = Drawing.new("Square")
                healthOutline.Filled = false
                healthOutline.Thickness = 1
                healthOutline.Color = Color3.new(0, 0, 0)
                healthOutline.Size = Vector2.new(barWidth, fullHeight)
                healthOutline.Position = Vector2.new(barX, minVector.Y)
                healthOutline.Visible = true
                table.insert(activeDrawings, healthOutline)

                -- Filled bar (green)
                local barHeight = fullHeight * healthPercent
                local healthBar = Drawing.new("Square")
                healthBar.Filled = true
                healthBar.Color = Color3.fromRGB(0, 255, 0)
                healthBar.Size = Vector2.new(barWidth, barHeight)
                healthBar.Position = Vector2.new(barX, minVector.Y)
                healthBar.Visible = true
                table.insert(activeDrawings, healthBar)
            end
        end

        ---- NAME ----
        if config.visual.name and headVisible then
            local nameText = Drawing.new("Text")
            nameText.Text = player.Name
            nameText.Size = 13
            nameText.Color = currentColor
            nameText.Position = Vector2.new(headScreenPos.X, headScreenPos.Y + 20)
            nameText.Center = true
            nameText.Visible = true
            table.insert(activeDrawings, nameText)
        end

        ---- LINE (tracer from top center to head) ----
        if config.visual.line and headVisible then
            local topCenter = Vector2.new(Camera.ViewportSize.X / 2, 0)
            local tracerLine = Drawing.new("Line")
            tracerLine.Thickness = 1
            tracerLine.Color = currentColor
            tracerLine.From = topCenter
            tracerLine.To = headScreenPos
            tracerLine.Visible = true
            table.insert(activeDrawings, tracerLine)
        end

        ---- PLAYER OUTLINE (Highlight) ----
        if config.visual.playerOutline then
            local existingHighlight = character:FindFirstChild("PotatoOutline")
            if not existingHighlight then
                local newHighlight = Instance.new("Highlight")
                newHighlight.Name = "PotatoOutline"
                newHighlight.FillTransparency = 1
                newHighlight.Parent = character
                existingHighlight = newHighlight
            end
            existingHighlight.OutlineColor = currentColor
        end
    end

    -- Remove highlights for players not drawn (out of range, dead, or not meeting conditions)
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and not drawnPlayers[player] then
            local character = getCharacter(player)
            if character then
                local highlight = character:FindFirstChild("PotatoOutline")
                if highlight then
                    highlight:Destroy()
                end
            end
        end
    end
end

-- Logo button (appears when minimized)
function createLogoButton(gui, mainWindow)
    if logoButton then return end
    local logoGui = Instance.new("ScreenGui")
    logoGui.DisplayOrder = 99
    logoGui.Parent = LocalPlayer:WaitForChild("PlayerGui")
    logoButton = Instance.new("TextButton")
    logoButton.Size = UDim2.new(0, 45, 0, 45)
    logoButton.Position = UDim2.new(0, mainWindow.AbsolutePosition.X + mainWindow.AbsoluteSize.X / 2 - 22.5, 0, mainWindow.AbsolutePosition.Y - 30)
    logoButton.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
    logoButton.BorderSizePixel = 0
    logoButton.Text = "🥔"
    logoButton.TextColor3 = Color3.fromRGB(0, 0, 0)
    logoButton.Font = Enum.Font.GothamBold
    logoButton.TextSize = 22
    logoButton.Parent = logoGui
    Instance.new("UICorner", logoButton).CornerRadius = UDim.new(0, 22)

    local dragStartPos, logoStartPos
    local isDragging = false
    local moved = false
    logoButton.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            isDragging = true
            moved = false
            dragStartPos = input.Position
            logoStartPos = logoButton.Position
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    isDragging = false
                    if not moved then
                        guiHidden = false
                        mainWindow.Visible = true
                        logoGui:Destroy()
                        logoButton = nil
                    end
                end
            end)
        end
    end)
    logoButton.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement and isDragging then
            local delta = input.Position - dragStartPos
            if delta.Magnitude > 5 then moved = true end
            if moved then
                logoButton.Position = UDim2.new(
                    logoStartPos.X.Scale, logoStartPos.X.Offset + delta.X,
                    logoStartPos.Y.Scale, logoStartPos.Y.Offset + delta.Y
                )
            end
        end
    end)
end

-- ================== FLY & NOCLIP ==================
local flyConnection = nil
local noclipConnection = nil

function startFly()
    local character = LocalPlayer.Character
    if not character then return end
    local humanoid = character:FindFirstChild("Humanoid")
    local rootPart = character:FindFirstChild("HumanoidRootPart")
    if not humanoid or not rootPart then return end
    humanoid.PlatformStand = true
    local bodyGyro = Instance.new("BodyGyro")
    bodyGyro.P = 9e9
    bodyGyro.MaxTorque = Vector3.new(9e9, 9e9, 9e9)
    bodyGyro.CFrame = rootPart.CFrame
    bodyGyro.Parent = rootPart
    local bodyVelocity = Instance.new("BodyVelocity")
    bodyVelocity.Velocity = Vector3.new(0, 0, 0)
    bodyVelocity.MaxForce = Vector3.new(9e9, 9e9, 9e9)
    bodyVelocity.Parent = rootPart
    flyConnection = RunService.Heartbeat:Connect(function()
        if not config.player.fly then return end
        local moveDirection = Vector3.new(0, 0, 0)
        if UserInputService:IsKeyDown(Enum.KeyCode.W) then moveDirection = moveDirection + Camera.CFrame.LookVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.S) then moveDirection = moveDirection - Camera.CFrame.LookVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.A) then moveDirection = moveDirection - Camera.CFrame.RightVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.D) then moveDirection = moveDirection + Camera.CFrame.RightVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.Space) then moveDirection = moveDirection + Vector3.new(0, 1, 0) end
        if UserInputService:IsKeyDown(Enum.KeyCode.LeftShift) then moveDirection = moveDirection - Vector3.new(0, 1, 0) end
        bodyVelocity.Velocity = moveDirection.Unit * config.player.flySpeed
        bodyGyro.CFrame = Camera.CFrame
    end)
end

function stopFly()
    if flyConnection then flyConnection:Disconnect(); flyConnection = nil end
    local character = LocalPlayer.Character
    if character then
        local rootPart = character:FindFirstChild("HumanoidRootPart")
        if rootPart then
            rootPart:FindFirstChild("BodyGyro"):Destroy()
            rootPart:FindFirstChild("BodyVelocity"):Destroy()
        end
        local humanoid = character:FindFirstChild("Humanoid")
        if humanoid then humanoid.PlatformStand = false end
    end
end

function startNoclip()
    local character = LocalPlayer.Character
    if not character then return end
    local rootPart = character:FindFirstChild("HumanoidRootPart")
    if not rootPart then return end
    local bodyForce = Instance.new("BodyForce")
    bodyForce.Force = Vector3.new(0, workspace.Gravity * rootPart.AssemblyMass, 0)
    bodyForce.Parent = rootPart
    noclipConnection = RunService.Heartbeat:Connect(function()
        if not config.player.noclip then return end
        for _, part in pairs(character:GetChildren()) do
            if part:IsA("BasePart") then part.CanCollide = false end
        end
    end)
end

function stopNoclip()
    if noclipConnection then noclipConnection:Disconnect(); noclipConnection = nil end
    local character = LocalPlayer.Character
    if character then
        local rootPart = character:FindFirstChild("HumanoidRootPart")
        if rootPart then
            local bodyForce = rootPart:FindFirstChild("BodyForce")
            if bodyForce then bodyForce:Destroy() end
        end
        for _, part in pairs(character:GetChildren()) do
            if part:IsA("BasePart") then part.CanCollide = true end
        end
    end
end

-- ================== AIMBOT LOOP ==================
RunService.Heartbeat:Connect(function()
    if fovCircle then
        fovCircle.Visible = config.aimbot.enabled
        fovCircle.Radius = config.aimbot.fov
        fovCircle.Position = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y / 2)
    end

    if not config.aimbot.enabled then return end
    if config.aimbot.mouseAim and not UserInputService:IsMouseButtonPressed(Enum.UserInputType.MouseButton1) then return end

    local target = findClosestPlayerToCursor(config.aimbot.fov)
    if not target then return end
    local character = getCharacter(target)
    if not character then return end
    local aimPart = character:FindFirstChild(config.aimbot.aimPart) or character:FindFirstChild("Head") or character:FindFirstChild("Torso") or character:FindFirstChild("HumanoidRootPart")
    if not aimPart then return end

    if config.aimbot.mouseAim then
        local screenPos, onScreen = worldToScreen(aimPart.Position)
        if onScreen and mousemoverel then
            local mousePos = UserInputService:GetMouseLocation()
            local deltaX = screenPos.X - mousePos.X
            local deltaY = screenPos.Y - mousePos.Y
            mousemoverel(deltaX * config.aimbot.mouseAimSmoothness / 10, deltaY * config.aimbot.mouseAimSmoothness / 10)
        end
    else
        Camera.CFrame = Camera.CFrame:Lerp(CFrame.new(Camera.CFrame.Position, aimPart.Position), config.aimbot.smoothness)
    end
end)
